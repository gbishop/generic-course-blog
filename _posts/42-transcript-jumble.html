Today we will write a jumble solver. If you want to play along, here is a list of words to use.

import time

Letters = raw_input('Enter your jumbled word. ')
Letters = sorted(list(Letters.lower().strip()))

start = time.time()
for line in file('/usr/share/dict/words', 'r'):
    word = line.strip()
    t = sorted(list(word.lower()))
    if Letters == t:
        print word
done = time.time()
print 'it took', done-start, 'seconds'
