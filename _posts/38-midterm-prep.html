def FB(n):
    r = []
    i = 1
    while i < n:
        r.append(i)
        i = i + 2
    return r

def FC(n):
    s = 0
    for i in range(1,n):
        if i % 3 != 0: # % is the mod operator, remainder after division
            s = s + i
    return s
print FC(5)	_________________________

def FD(n):
    r = []
    for x in range(n):
        for y in range(x+1,n):
            r.append((x,y))
    return r
print FD(4)	_________________________

def FE(n):
    e = 1
    f = 0
    while e < n:
        e = e+f
        f = e
    return e
print FE(6)	_________________________
