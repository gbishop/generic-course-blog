M = np.array([ [1.5, 2.3, 3.1], [1.8, 2.9, 3.3], [1.7, 2.2, 3.9] ])
V = np.array([ 14, 1, 23 ])

print '#1'
for i in M:
    print 'i=', i

print '#2'
for x in M:
    print 'x=', x

print '#3'
for i,x in enumerate(M):
    print 'i=',i,'x=',x

print '#4'
for i in range(len(M)):
    print 'i=', i

print '#5'
for i in V:
    print 'i=', i

print '#6'
for x in V:
    print 'x=', x

print '#7'
for i,x in enumerate(V):
    print 'i=',i,'x=',x

print '#8'
for i in range(len(V)):
    print 'i=', i
